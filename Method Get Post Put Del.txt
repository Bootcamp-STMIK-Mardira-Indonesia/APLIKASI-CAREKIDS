Method Get

import { HttpClient } from '@angular/common/http';

constructor(private http: HttpClient) {}

getData() {
  return this.http.get('https://api.example.com/data');
}

ngOnInit() {
  this.getData().subscribe(data => {
    console.log(data);
  });
}


Method Post 

postData(data) {
  return this.http.post('https://api.example.com/data', data);
}

ngOnInit() {
  this.postData({ name: 'John Doe' }).subscribe(data => {
    console.log(data);
  });
}


Method Put

updateData(data) {
  return this.http.put('https://api.example.com/data/1', data);
}

ngOnInit() {
  this.updateData({ name: 'Jane Doe' }).subscribe(data => {
    console.log(data);
  });
}


Method Del

deleteData(id) {
  return this.http.delete(`https://api.example.com/data/${id}`);
}

ngOnInit() {
  this.deleteData(1).subscribe(data => {
    console.log(data);
  });
}


consume API Login

login(username: string, password: string) {
  return this.http.post<any>(`https://api.example.com/login`, { username, password })
    .pipe(map(user => {
      // simpan token JWT pada local storage
      localStorage.setItem('access_token', user.access_token);
      return user;
    }));
}


getData() {
  let headers = new HttpHeaders({
    'Authorization': 'Bearer ' + localStorage.getItem('access_token')
  });

  return this.http.get<any>(`https://api.example.com/data`, { headers });
}
